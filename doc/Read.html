def call(*args, env:, frames:)
  vars = env[:in].gets.chomp.split(frames[:ifs])
  combine = args.map(&:to_sym).zip vars
  combine.each do |e|
    frames[e[0]] = e[1]
  end
  frames.merge
  true
end
