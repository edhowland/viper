def call(env:, frames:)
  my_env = env._clone
  sio = StringIO.new
  my_env[:out] = sio
  super env: my_env, frames: frames
  sio.close_write
  sio.rewind
  result = sio.read.tr("\n", ' ').split
  if result.length == 1
    result[0]
  else
    result
  end
end
