def make object
  facade = mk_facade object.class
  unless facade.nil?
    facade.new(object)
  else
    StringIOFacade.new(StringIO.new(''))
  end
end

def mk_facade klass

    BufNode => BufWriteFacade,
    Array => ArrayFacade
  }[klass]
