def call(*args, env:, frames:)
  result = true
  perform(args[0], env: env, frames: frames) do |node|
    unless node.respond_to? :empty?
      env[:err].puts "cannot check if #{args[0]} is empty"
      return false
    end
    result = node.empty?
    ''
  end
  result
end
