def [] sym
  result = @frames.reduce('') do |i, j|
  if j.has_key? sym
      j[sym]
    else
      i
    end
  end
  result  #.to_s
end

def _clone
  nframes = @frames.map {|e| e.clone }
  nfunctions = @functions.clone
  naliases = @aliases.clone
  nfs = FrameStack.new(frames:nframes, functions:nfunctions, aliases:naliases)
  nfs.vm = @vm
  nfs
end
