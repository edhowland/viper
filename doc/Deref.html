def call frames:, env:{}
  resolve! frames:frames
  var = @value
  case var
  when String
    if range?
      return handle_range
    end
  result = var.split(frames[:ifs])
  if result.length > 1
    return result
  else
    return result[0]
  end
  else
    return var
  end
end
