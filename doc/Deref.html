def call(frames:, env: {})
  resolve! frames: frames
  var = @value
  case var
  when String
    return handle_range if range?
    result = var.split(frames[:ifs])
    return head_or_all(result)
  else
    return var
  end
end

def range?
  @matches = @value.match(/([^.]+)\.\.(.+)/)
  !@matches.nil?
end
