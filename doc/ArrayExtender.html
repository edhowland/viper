def cut(limit)
  value = copy(limit)
  if limit >= 0
    slice!(0..(limit - 1))
  else
    slice!(limit..-1)
  end
  value
end

def last_line
  return '' if self[-1] == "\n"
  return to_s if rcount_nl == length
  self[-(rcount_nl - 1)..-1].join('')
end

def rword_index
  offset = to_s.rindex(/\s|\n/)
  offset = to_s.rindex(/^\w/) if offset.nil?
  (offset.nil? ? '' : self[offset..-1].join('').lstrip)
end
