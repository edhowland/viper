def associate(fname)
  dir = fname.pathmap('%d/')
  file = fname.pathmap('%f')
  ext = fname.pathmap('%x')

  (match_file(file) || match_ext(ext) || match_dir(dir) || :default)
end

def dir(pattern, sym)
  regex = /%r{(.*)}/
  m = regex.match(pattern)
  unless m.nil?
    s_regex = Regexp.new(m[1])
    dir_regex s_regex, sym
  else
    dir_lit pattern, sym
  end
end

def match_regex(regexs, string)
  found_a = regexs.keys.map { |r| r.match(string) }.reject(&:nil?).sort { |a, b| b.to_s.length <=> a.to_s.length }
  unless found_a.empty?
    regex = found_a.first.regexp
    return regexs[regex]
  end
  nil
end
