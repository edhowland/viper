def call(*args, env:, frames:)
  super do |*_a|
    method = :puts
    if @options[:n]
      method = :print # if args[0] == '-n'
    end
    if @options[:w]
      @out.send(method, @in.read.split(/\b/)
        .map(&:strip).reject(&:empty?).length)
    elsif @options[:l]
      @out.puts @in.read.each_line.to_a.length
    else
      @out.send(method, @in.read.length)
    end
  end
end
