def expand
return @storage if @storage.empty?

  @storage.reject! {|e| ListType === e }

  @storage = @storage.map(&:expand).flatten.map {|e| @links[e] || e }
      @storage.each {|e| e.extend Toppable }
  @storage.first.top = true
  @storage
end
