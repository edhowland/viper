def write string
  enum = string.each_line
  line = @io['line']
  begin
  line.right = enum.next
  node = @io
    loop do
      contents = enum.next
      node = node.mknode 'nl'
      node['line'].right = contents
    end
  rescue StopIteration
    #
  end
end
