def call env:, frames:
  @vm = frames.vm
  save_pwd @vm
  vm = @vm._clone
  local_vars = frames
  local_vars.push
  local_ios = env
  local_ios.push
  @redirections.each {|e| redirect(e, env:local_ios, frames:local_vars) }
  vm.ios = local_ios
  vm.fs = local_vars
  closers = open_redirs env:local_ios
  begin
    vm.call @block
  rescue VirtualMachine::BreakCalled => err
    return true
  ensure
    restore_pwd @vm
    close_redirs closers
    local_vars.pop
    local_ios.pop
  end
end

def line_number
  result = 0

  end
  result
end

end
