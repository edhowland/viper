def insnode text=''
  child = @list['nl']
  unless child
    mknode ''
  child = @list['nl']
  end
  baby = mknode ''
  lnode = LineNode.new(baby, 'line')
  lnode.right = text
  baby['line'] = lnode
  baby.parent = child.parent
  child.parent = baby
  baby['nl'] = child
  baby
end

def newline
  exist = @list['nl']
  nl = BufNode.new(self, 'nl')
  nl['nl'] = exist unless exist.nil?
  exist.parent = nl unless exist.nil?
  @list['nl'] = nl
end
