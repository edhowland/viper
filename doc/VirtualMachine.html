def initialize  env:FrameStack.new(frames:[{in: $stdin, out: $stdout, err: $stderr}]), frames:FrameStack.new
  @fs= frames
  @fs[:exit_status] = true
  @fs.vm = self
  @fs[:pwd] = Hal.pwd
  @fs[:vhome] = PhysicalLayer.realpath(Hal.dirname(__FILE__) + '/../..')
  @fs[:prompt] = 'vish >'
  @fs[:oldpwd] = Hal.pwd
  @fs[:version] = Vish::VERSION
  @fs[:ifs] = " "
  @fs[:ofs] = ' '
  @fs[:_debug] = false
  @ios = env
  @seen = [] # seen aliases during dealias
end

def _chdir path
  @fs.first[:oldpwd] =  @fs[:pwd]
  result = true
  begin
    Hal.chdir path, @fs[:pwd]
    rescue Errno::ENOENT => exc
      raise exc
  rescue => err
    result = false
  ensure
    @fs.first[:pwd] = Hal.pwd
  end
  result
end

def _return *args, env:, frames:
  if args.empty?
    code = true
  elsif args[0] == 'false'
      code = false
    elsif args[0] == 'true'
      code = true
    else
      code = args[0]
  end
  raise VirtualMachine::ReturnCalled.new code
end

def cd *args, env:, frames:
  begin
    if !args.empty? && args[0] == '-'
      oldpwd = @fs[:oldpwd]
      self._chdir oldpwd
      self.pwd *args, env:env, frames:frames
    elsif args.empty?
      # go back to :proj
      self._chdir frames[:proj]
      self.pwd *args, env:env, frames:frames
    else
      self._chdir args[0]
    end
    true
  rescue Errno::ENOENT => err
    env[:err].puts "cd: #{err.message}"
    false
  end
end

def source *args, env:, frames:
  if args.empty?
    env[:err].puts 'source: missing argument'
    false
  else
  begin
      block = Visher.parse!(File.read(args[0]))
      @fs.push
      @fs[:__FILE__] = args[0]
      self.call block
      @fs.pop
    rescue ExitCalled
      raise ExitCalled
    rescue ReturnCalled
      # nop
    rescue => err
      line = 0
      if err.respond_to? :line_number
        line = err.line_number
      end
      env[:err].puts "#{args[0]}:#{line}: exception #{err.message}"
      false
    end
  end
end
