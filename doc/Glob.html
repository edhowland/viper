def call(env:, frames:)
  derefed_pattern = @pattern.call frames: frames
  result = []
  result = Hal[derefed_pattern] if derefed_pattern =~ /\*/
  return derefed_pattern if result.empty?
  return result[0] if result.length == 1
  result
end
