string_buffer.rb - class StringBuffer StringBuffer - internal class for Buffer. Wrapps String class and makes it look like an Array with push, pop, shift and unshift methods.

def cut(limit)
  value = copy(limit)
  @buffer = if limit < 0
              @buffer[0..(limit - 1)]
            else
              @buffer[limit..-1]
            end
  value
end

def pop
  raise BufferExceeded.new('Exceeded back') if @buffer.empty?
  result = @buffer[-1]
  @buffer = @buffer[0..-2]
  result
end

def rword_index
  offset = @buffer.rindex(/\s|\n/)

  offset = @buffer.rindex(/^\w/) if offset.nil?
  (offset.nil? ? '' : @buffer[offset..-1].lstrip)
end
