def parse!(args = [])
  value = []
  ments = args.map { |e| @flags[e].nil? ? e : @flags[e] }
  iter = ments.each
  begin
    loop do
      arg = iter.next
      if arg.instance_of?(Proc)
        case arg.arity
        when 0
          arg.call
        when 1
          param = iter.next
          arg.call param
        else
          raise WrongArgCount, arg.arity
        end
      else
        value << arg
      end
    end
  rescue
    StopIteration
  end
  value
end
