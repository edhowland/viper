def call(*args, env:, frames:)
  super do |*a|
    pattern = a.shift
    regex = regexp pattern
    result = false
    @in.read.each_line do |l|
      matches = l.match regex
      next unless matches
      result = true
      if @options[:o]
        say matches[0]
      else
        say l
      end
    end
    result
  end
end

def regexp(string)
  if string[0] == '/'
    Regexp.new string[1..-2]
  else
    Regexp.new string
  end
end
