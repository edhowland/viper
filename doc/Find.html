def call *args, env:, frames:
  @source, @filter, @exec = args.shift(3)
  case @filter
  when String
    @filter = Selector.new(@filter)
  when Lambda
    # nop
  else
    env[:err].puts 'find: second arg must be string or lambda'
    return false
  end
  @exec ||= Echo.new
  begin
    Hal['**'].select {|e| @filter.call(e, env:env, frames:frames) }.each {|e| @exec.call(e, env:env, frames:frames) }
  rescue VirtualMachine::BreakCalled => err
    return true
  end
  true
end
