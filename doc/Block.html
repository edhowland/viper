def call env:, frames:
  @statement_list.each do |s|
    begin
      #Event.trigger *(s.to_s.split), env:env, frames:frames
      Event.on(*(s.to_s.split),env:env, frames:frames)
      s.call env:env, frames:frames 
    rescue => err
      err.extend LineNumberable
      err.line_number = s.line_number
      raise err
    end
  end
  frames[:exit_status]
end
