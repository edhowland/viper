def _chdir(elements, start = @wd)
  spath = path_from_start_elements(start.name, elements)
  elements.each do |e|
    start = if start.nil? || start[e].nil?
              nil
            else
              start[e]
            end
  end
  raise Errno::ENOENT, spath if start.nil?
  @wd = start
end

def _mkdir(elements, start = @root)
  node = start
  elements.each do |e|
    node = if node[e].nil?
             node.mknode e
           else
             node[e]
           end
  end
end

def list(path = '.')
  if path == '.'
    @wd.keys
  else
    start, *elements = path_to_elements path
    mynode = node(elements, start)
    if VFSNode === mynode
      mynode.keys
    else
      [elements[-1]]
    end
  end
end

def parents(node)
  rpath = []
  until node.parent.nil?
    rpath << node
    node = node.parent
  end
  rpath.reverse
end

def path_to_elements(path)
  elements = path.split('/')
  if elements[0].nil? || elements[0].empty?
    elements[0] = @root
  else
    elements.unshift @wd
  end
  elements
end

def pwd
  pathr = []
  p = @wd
  until p.parent.nil?
    pathr << p
    p = p.parent
  end
  '/' + pathr.reverse.map(&:name).join('/')
end
