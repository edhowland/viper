def _chdir elements, start=@wd
  spath = path_from_start_elements(start.name, elements)
  elements.each do |e|
    if start.nil? || start[e].nil?
      start = nil
    else
      start = start[e]
    end
  end
  raise Errno::ENOENT.new(spath) if start.nil? 
  @wd = start
end

def _mkdir elements, start=@root
    node = start
    elements.each do |e|
      if node[e].nil?
        node = node.mknode e
      else
        node = node[e]
      end
  end
end

def parents node
  rpath = []
  until node.parent.nil?
    rpath << node
    node = node.parent
  end
  rpath.reverse
end

def path_to_elements path
  elements = path.split('/')
  if elements[0].nil? || elements[0].empty?
    elements[0] = @root
  else
  elements.unshift @wd
end
elements
end

def pwd
  pathr = []
  p = @wd
  until p.parent.nil?
    pathr << p
    p = p.parent
  end
  '/' + pathr.reverse.map {|e| e.name }.join('/')
end
