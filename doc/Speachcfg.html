def seterr arg, env:
if arg == 'stderr'
    env.first[:err] = $stderr
  elsif arg == 'EspeakIF'
    env.first[:err] = EspeakIF.new
  else
    env[:err].puts "cannot set speach engine to #{arg}. Unknown class name"
  end
end
